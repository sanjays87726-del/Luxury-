using System.IO;
using UnityEngine;

namespace RealDriveCity.SaveSystem
{
    [System.Serializable]
    public class SaveData
    {
        public Vector3 playerPosition;
        public Quaternion playerRotation;
        public string currentVehicleId;
        public float currentFuel;
        public float gameTimeOfDay; // 0-24h
        public int graphicsQuality;
        public string[] unlockedVehicleIds;
        public long lastSavedUnixTime;
    }

    /// <summary>
    /// Lightweight offline JSON save system — no internet/cloud dependency.
    /// Writes to Application.persistentDataPath, which survives app updates on Android.
    /// </summary>
    public static class SaveManager
    {
        private static readonly string SavePath = Path.Combine(Application.persistentDataPath, "save_slot_1.json");

        public static void Save(SaveData data)
        {
            data.lastSavedUnixTime = System.DateTimeOffset.UtcNow.ToUnixTimeSeconds();
            string json = JsonUtility.ToJson(data, true);
            try
            {
                File.WriteAllText(SavePath, json);
            }
            catch (IOException e)
            {
                Debug.LogError($"[SaveManager] Failed to write save file: {e.Message}");
            }
        }

        public static SaveData Load()
        {
            if (!File.Exists(SavePath)) return null;
            try
            {
                string json = File.ReadAllText(SavePath);
                return JsonUtility.FromJson<SaveData>(json);
            }
            catch (IOException e)
            {
                Debug.LogError($"[SaveManager] Failed to read save file: {e.Message}");
                return null;
            }
        }

        public static bool HasSave() => File.Exists(SavePath);

        public static void DeleteSave()
        {
            if (File.Exists(SavePath)) File.Delete(SavePath);
        }
    }
}
